---方法一 使用Docker Hub---

--1注册docker hub账号

--2docker hub上创建仓库

--3docker 本地登录
docker login

--4将容器commit 成镜像
docker tag <existing-image> <hub-user>/<repo-name>[:<tag>]
docker tag relayserver leon08162001/tayarelayserver:latest

--5docker push 镜像到docker hub 的仓库
docker push <hub-user>/<repo-name>:<tag>
docker push leon08162001/tayarelayserver:latest


--6命令验证
docker inspect leon08162001/tayarelayserver:latest

---方法二 使用Azure Container Registry---

--登入您的 Azure Container Registry
docker login <login-server>
docker login imagerelayserver1.azurecr.io

--azure容器登錄(標記 Docker Image)
docker tag relayserver:latest <registry-name>.azurecr.io/reservationsystem:latest
docker tag relayserver:latest imagerelayserver1.azurecr.io/relayserver:latest

--將映像上傳至Azure Container Registry
docker push <registry-name>.azurecr.io/reservationsystem:latest
docker push imagerelayserver1.azurecr.io/relayserver:latest
